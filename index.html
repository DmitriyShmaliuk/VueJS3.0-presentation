Other features

  1. Script setup
  2. Script vars

Script setup

<script>
            import { ref } from "vue";
            export default {
              setup() {
                const isShow = ref(false);
                return {
                  isShow,
                };
              },
            };
            </script>
          
            <script setup>
              import { ref } from "vue";
              export const isShow = ref(false);
            </script>

Script vars

            <template>
              <div class="text">hello</div>
            </template>

            export default {

                }
              }

             </script>
